public int numIslands(char[][] grid) {
    int count = 0;
    // йҒҚеҺҶеӣҫ
    for (int i = 0; i < grid.length; i++) {
        for (int j = 0; j < grid[0].length; j++) {
            // еҪ“йҒҚеҺҶзӮ№еҖјдёә1ж—¶
            if (grid[i][j] == '1') {
                // еІӣеұҝж•°йҮҸеҠ дёҖ
                count++;
                // dfsжҗңзҙўжүҖжңүзӣёиҝһзҡ„зӮ№
                dfs(grid, i, j);
            }
        }
    }
    return count;
}

// еқҗж ҮдёҠзҡ„еӣӣдёӘж–№еҗ‘(0, 1), (0, -1), (1, 0), (-1, 0)
private int[] dx = {0, 0, 1, -1};
private int[] dy = {1, -1, 0, 0};

private void dfs(char[][] grid, int i, int j) {
    // еҲӨж–ӯжҳҜеҗҰжҳҜжңүж•ҲзӮ№
    if (!valid(grid, i, j))   return;
    // е°Ҷиө°иҝҮзҡ„зӮ№ж•°еҖјж”№дёә0пјҢйҒҝе…ҚйҮҚеӨҚи®ҝй—®
    grid[i][j] = '0';
    // еҗ‘еӣӣдёӘж–№еҗ‘йҖ’еҪ’и°ғз”Ёdfs
    for (int k = 0; k < 4; k++)
        dfs(grid, i + dx[k], j + dy[k]);
}

private boolean valid(char[][] grid, int i, int j) {
    // жңүж•ҲзӮ№йңҖиҰҒеқҗж ҮеҗҲжі•пјҢдё”еҖјдёә1
    return i >= 0 && j >= 0 && i < grid.length && j < grid[0].length && grid[i][j] == '1';
}

public boolean exist(char[][] board, String word) {
    if (word == null || word.length() == 0) return true;

    // йҒҚеҺҶboardжҜҸдёӘзӮ№
    for (int i = 0; i < board.length; i++)
        for (int j = 0; j < board[0].length; j++)
            // еҰӮжһңеҪ“еүҚеӯ—з¬ҰдёҺwordејҖеӨҙеӯ—жҜҚзӣёеҗҢпјҢжү§иЎҢdfs
            if (board[i][j] == word.charAt(0)) {
                // дҪҝз”Ё2з»ҙvisitedж•°з»„ж Үи®°и®ҝй—®иҝҮзҡ„дҪҚзҪ®
                boolean[][] visited = new boolean[board.length][board[0].length];
                // dfsйҖ’еҪ’зҡ„еҢ№й…Қboard[i][j]дёҺwordз¬¬kдҪҚзҡ„еӯ—жҜҚ
                if (dfs(board, i, j, word, 0, visited))
                    return true;
            }
    return false;
}

// еқҗж ҮдёҠзҡ„еӣӣдёӘж–№еҗ‘(0, 1), (0, -1), (1, 0), (-1, 0)
private final int[] dx = {0, 0, 1, -1};
private final int[] dy = {1, -1, 0, 0};

private boolean dfs(char[][] board, int i, int j, String word, int k, boolean[][] visited) {
    if (k == word.length()) // еҰӮжһңеҢ№й…ҚеҲ°иҫҫеҚ•иҜҚз»“е°ҫпјҢиҜҙжҳҺеҢ№й…ҚжҲҗеҠҹ
        return true;
    if (invalid(board, i, j, word, k, visited))  // еҲӨж–ӯжҳҜеҗҰжҳҜжңүж•ҲзӮ№
        return false;

    // е°ҶеҪ“еүҚдҪҚзҪ®ж Үи®°дёәе·Із»Ҹи®ҝй—®иҝҮ
    visited[i][j] = true;
    for (int c = 0; c < 4; c++) // еҗ‘еӣӣдёӘж–№еҗ‘йҖ’еҪ’и°ғз”Ёdfs
        if (dfs(board, i + dx[c], j + dy[c], word, k + 1, visited))
            return true;
    // зҠ¶жҖҒеӣһжәҜпјҢе°ҶзЁӢеәҸзҠ¶жҖҒеӣһеӨҚжҲҗжІЎжңүи®ҝй—®еҪ“еүҚдҪҚзҪ®зҡ„зҠ¶жҖҒ
    visited[i][j] = false;
    return false;
}

private boolean invalid(char[][] board, int i, int j, String word, int k, boolean[][] visited) {
    // жңүж•ҲзӮ№йңҖиҰҒеқҗж ҮеҗҲжі•пјҢжІЎжңүи®ҝй—®иҝҮпјҢдё”board[i][j]дёҺwordз¬¬kдҪҚзҡ„еӯ—жҜҚзӣёеҗҢ
    return i < 0 || j < 0 || i >= board.length || j >= board[0].length
            || visited[i][j] || word.charAt(k) != board[i][j];
}

жҖқи·ҜеҲҶжһҗ1

жң¬йўҳзҡ„еӣҫжІЎжңүзӣҙжҺҘз»ҷеҮәпјҢйңҖиҰҒжҲ‘д»¬ж №жҚ®з»ҷеҮәзҡ„еҸҳжҚўи§„еҲҷиҮӘе·ұе»әжЁЎгҖӮжҲ‘д»¬еҸҜд»Ҙе°ҶжҜҸдёӘеҚ•иҜҚи§ҶдёәдёҖдёӘиҠӮзӮ№пјҢеҸҜд»Ҙдә’зӣёиҪ¬жҚўзҡ„дёӨдёӘеҚ•иҜҚд№Ӣй—ҙеӯҳеңЁдёҖжқЎиҫ№гҖӮжүҖд»Ҙйўҳзӣ®еҸҜд»ҘиҪ¬жҚўдёәпјҢеҜ»жүҫд»Һиө·е§ӢзӮ№startиө°еҲ°з»ҲзӮ№endзҡ„жңҖзҹӯи·Ҝеҫ„пјҲжҜҸжқЎиҫ№й•ҝеәҰзӣёзӯүеқҮдёә1пјүгҖӮеҜ№дәҺжҜҸжқЎиҫ№зӯүй•ҝзҡ„еӣҫпјҢеҜ»жүҫжңҖзҹӯи·Ҝеҫ„зҡ„еҹәжң¬ж–№жі•жҳҜдҪҝз”ЁBFSгҖӮ

  1. ж–°е»әдёҖдёӘйҳҹеҲ—е’ҢдёҖдёӘйӣҶеҗҲпјҢд»Һиө·зӮ№startејҖе§Ӣжү§иЎҢbfs
  2. жҜҸдёҖиҪ®йҒҚеҺҶеҪ“еүҚйҳҹеҲ—дёӯжүҖжңүиҠӮзӮ№гҖӮеҜ№жҜҸдёӘиҠӮзӮ№пјҢе°ҶиҜҘиҠӮзӮ№иғҪиө°еҲ°зҡ„жүҖжңүиҠӮзӮ№еҠ е…ҘйҳҹеҲ—
  3. еҰӮжһңйҒҮеҲ°endиҠӮзӮ№пјҢйҖҖеҮәbfsпјӣеҗҰеҲҷе°Ҷи·Ҝеҫ„й•ҝеәҰ+1гҖӮзӣҙеҲ°йҳҹеҲ—дёәз©әгҖӮ

д»Јз Ғе®һзҺ°1

public int ladderLength(String beginWord, String endWord, List<String> wordList) {
    // е°ҶlistиҪ¬жҲҗsetпјҢж–№дҫҝеҝ«йҖҹжҹҘжүҫ
    Set<String> words = new HashSet<>(wordList);
    if (!words.contains(endWord))
        return 0;

    // ж–°е»әдёҖдёӘйҳҹеҲ—е’ҢдёҖдёӘйӣҶеҗҲ
    Queue<String> queue = new LinkedList<>();
    Set<String> visited = new HashSet<>();
    // д»Һиө·зӮ№ејҖе§Ӣжү§иЎҢbfs
    queue.offer(beginWord);
    visited.add(beginWord);

    int step = 0;
    while (!queue.isEmpty()) {
        int size = queue.size();
        step++;
        // йҒҚеҺҶеҪ“еүҚйҳҹеҲ—дёӯжүҖжңүиҠӮзӮ№
        for (int i = 0; i < size; i++) {
            String curt = queue.poll();
            if (curt.equals(endWord)) // еҰӮжһңйҒҮеҲ°зӣ®ж ҮиҠӮзӮ№пјҢйҖҖеҮәbfs
                return step;
            // иҺ·еҫ—еҪ“еүҚиҠӮзӮ№иғҪиө°еҲ°зҡ„жүҖжңүиҠӮзӮ№
            List<String> neighbors = expanding(curt.toCharArray());
            for (String neighbor : neighbors) {
                if (words.contains(neighbor) && !visited.contains(neighbor)) {
                    // е°ҶжңӘи®ҝй—®зҡ„жңүж•ҲиҠӮзӮ№еҠ е…ҘйҳҹеҲ—
                    queue.offer(neighbor);
                    visited.add(neighbor);
                }
            }
        }
    }
    return 0;
}

private List<String> expanding(char[] word) {
    List<String> neighbors = new ArrayList<>();
    // еңЁеҚ•иҜҚзҡ„жҜҸдёҖдёӘдҪҚзҪ®
    for (int i = 0; i < word.length; i++) {
        char tmp = word[i];
        // з”Ёa-zеҫӘзҺҜжӣҝжҚўиҜҘдҪҚзҪ®зҡ„еӯ—з¬Ұ
        for (char j = 'a'; j <= 'z'; j++) {
            word[i] = j;
            neighbors.add(new String(word));
        }
        word[i] = tmp;
    }
    return neighbors;
}

д»Јз Ғе®һзҺ°2

public int ladderLength(String start, String end, List<String> wordList) {
    Set<String> dict = new HashSet<>(wordList);
    if (!dict.contains(end)) return 0;

    // дҪҝз”ЁдёӨдёӘйҳҹеҲ—
    Queue<String> startQueue = new LinkedList<>(), endQueue = new LinkedList<>();
    Set<String> startSet = new HashSet<>(), endSet = new HashSet<>();

    // еҲҶеҲ«е°Ҷstartе’Ңendж·»еҠ иҝӣеҜ№еә”зҡ„йҳҹеҲ—
    addWord(startQueue, startSet, start);
    addWord(endQueue, endSet, end);

    int step = 1;
    while (!startQueue.isEmpty() && !endQueue.isEmpty()) {
        // д»Һиҫғе°Ҹзҡ„йӮЈдёӘйҳҹеҲ—ејҖе§Ӣжү§иЎҢBFS
        boolean forwarding = startQueue.size() <= endQueue.size();
        Queue<String> queue = forwarding ? startQueue : endQueue;
        Set<String> forward = forwarding ? startSet : endSet, backward = forwarding ? endSet : startSet;

        step++;
        // йҒҚеҺҶеҪ“еүҚйҳҹеҲ—дёӯжүҖжңүиҠӮзӮ№
        int size = queue.size();
        for (int i = 0; i < size; i++){
            String curt = queue.poll();
            // иҺ·еҫ—еҪ“еүҚиҠӮзӮ№иғҪиө°еҲ°зҡ„жүҖжңүиҠӮзӮ№
            List<String> neighbors = expanding(curt.toCharArray());
            for (String neighbor : neighbors) {
                // жҺ’йҷӨдёҚеңЁеӯ—е…ёдёӯзҡ„иҠӮзӮ№
                if (!dict.contains(neighbor))
                    continue;
                // еҰӮжһңжҗңзҙўиҢғеӣҙдёҺеҸҰдёҖдҫ§йҳҹеҲ—зӣёйҒҮпјҢйҖҖеҮәbfs
                if (backward.contains(neighbor))
                    return step;
                // е°ҶжңӘи®ҝй—®зҡ„жңүж•ҲиҠӮзӮ№еҠ е…ҘйҳҹеҲ—
                if (!forward.contains(neighbor))
                    addWord(queue, forward, neighbor);
            }
        }
    }
    return 0;
}

private void addWord(Queue<String> queue, Set<String> set, String word) {
    // е°ҶеҚ•иҜҚеҠ е…ҘйҳҹеҲ—е’ҢйӣҶеҗҲпјҢйҳІжӯўйҮҚеӨҚи®ҝй—®
    queue.offer(word);
    set.add(word);
}

private List<String> expanding(char[] word) {
    List<String> neighbors = new ArrayList<>();
    // еңЁеҚ•иҜҚзҡ„жҜҸдёҖдёӘдҪҚзҪ®
    for (int i = 0; i < word.length; i++) {
        char tmp = word[i];
        // з”Ёa-zеҫӘзҺҜжӣҝжҚўиҜҘдҪҚзҪ®зҡ„еӯ—з¬Ұ
        for (char j = 'a'; j <= 'z'; j++) {
            word[i] = j;
            neighbors.add(new String(word));
        }
        word[i] = tmp;
    }
    return neighbors;
}

4.жЎҲдҫӢ: еҚ•иҜҚжҺҘйҫҷ2

з»ҷеҮәдёӨдёӘеҚ•иҜҚпјҲstartе’Ңendпјүе’ҢдёҖдёӘеӯ—е…ёпјҢжүҫеҮәд»ҺstartеҲ°endзҡ„жңҖзҹӯиҪ¬жҚўеәҸеҲ—пјҢиҫ“еҮәжңҖзҹӯеәҸеҲ—гҖӮ еҸҳжҚўи§„еҲҷеҰӮдёӢпјҡ + жҜҸж¬ЎеҸӘиғҪж”№еҸҳдёҖдёӘеӯ—жҜҚ + еҸҳжҚўиҝҮзЁӢдёӯзҡ„дёӯй—ҙеҚ•иҜҚеҝ…йЎ»еңЁеӯ—е…ёдёӯеҮәзҺ°

иҫ“е…Ҙ: start = "hit"пјҢend = "cog"пјҢwordList = ["hot","dot","dog","lot","log"] иҫ“еҮә: "hit"->"hot"->"dot"->"dog"->"cog"

жҖқи·ҜеҲҶжһҗ

жң¬йўҳжҳҜдёҠдёҖйўҳзҡ„иҝӣйҳ¶зүҲпјҢдёҠдёҖйўҳд»…д»…жҳҜиҰҒжұӮи®Ўз®—еҮәжңҖзҹӯеәҸеҲ—зҡ„й•ҝеәҰпјҢиҖҢжң¬йўҳиҰҒжұӮиҝ”еӣһжңҖзҹӯеәҸеҲ—жң¬иә«гҖӮ

з¬¬дёҖйғЁеҲҶи§Јжі•дёҺдёҠдёҖйўҳзұ»дјјпјҡдҪҝз”ЁbfsжҗңзҙўжңҖзҹӯи·Ҝеҫ„пјҢеҗҢж—¶е°ҶжҜҸдёҖжӯҘжҗңзҙўзҡ„з»“жһңзј“еӯҳдёӢжқҘпјҢи®°еҪ•иҜҘиҠӮзӮ№еҲ°startзҡ„жңҖзҹӯи·қзҰ»гҖӮиҝҷж ·жҲ‘д»¬е°ұжңүдәҶдёҖдёӘеӯ—е…ёзҡ„еӯҗйӣҶпјҢе…¶дёӯеҢ…еҗ«дәҶжүҖжңүеҸҜиғҪзҡ„еҸҳжҚўеәҸеҲ—гҖӮ

з¬¬дәҢйғЁеҲҶжҲ‘д»¬дҪҝз”ЁdfsпјҢд»ҺendеҮәеҸ‘пјҢеңЁдёҠдёҖжӯҘзј“еӯҳзҡ„еӯҗйӣҶдёӯеҗ‘startеӣһжәҜгҖӮ

д»Јз Ғе®һзҺ°

public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
    // е°ҶlistиҪ¬жҲҗsetпјҢж–№дҫҝеҝ«йҖҹжҹҘжүҫ
    Set<String> dict = new HashSet<>(wordList);
    dict.add(beginWord);

    List<List<String>> ans = new ArrayList<>();
    if (!dict.contains(endWord))
        return ans;

    // е»әз«ӢдёҖдёӘmapз”ЁдәҺзј“еӯҳжҜҸдёҖжӯҘжҗңзҙўзҡ„з»“жһң
    Map<Integer, Set<String>> map = new HashMap<>();
    // д»Һиө·зӮ№ејҖе§Ӣжү§иЎҢbfsпјҢиҝ”еӣһжңҖзҹӯи·Ҝеҫ„зҡ„й•ҝеәҰ
    int distance = bfs(beginWord, endWord, dict, map);

    // еҰӮжһңdistanceдёә0иҜҙжҳҺдёҚеӯҳеңЁжңҖзҹӯи·Ҝеҫ„
    if (distance == 0)
        return ans;
    // з”ЁеҲ—иЎЁи®°еҪ•жңҖзҹӯи·Ҝеҫ„
    List<String> path = new ArrayList<>();
    // д»ҺendеҮәеҸ‘
    path.add(endWord);
    // distanceд»ЈиЎЁеҪ“еүҚиҠӮзӮ№еҲ°startзҡ„и·қзҰ»пјҢд»Һdistance - 1ејҖе§Ӣеҗ‘startжҗңзҙў
    dfs(endWord, distance - 1, map, path, ans);
    for (List<String> list : ans) // е°ҶжүҖжңүжүҫеҲ°зҡ„и·Ҝеҫ„зҝ»иҪ¬жҲҗд»ҺstartеҲ°end
        Collections.reverse(list);
    return ans;
}

// дёҺжЎҲдҫӢ3зҡ„и§Јжі•1еҹәжң¬зӣёеҗҢ
private int bfs(String beginWord, String endWord, Set<String> dict, Map<Integer, Set<String>> map) {
    // ж–°е»әдёҖдёӘйҳҹеҲ—е’ҢдёҖдёӘйӣҶеҗҲ
    Queue<String> queue = new LinkedList<>();
    Set<String> visited = new HashSet<>();

    // д»Һиө·зӮ№ејҖе§Ӣжү§иЎҢbfs
    queue.add(beginWord);
    visited.add(beginWord);

    int distance = 0;
    while (!queue.isEmpty()) {
        int size = queue.size();
        distance++;
        map.put(distance, new HashSet<>());
        // йҒҚеҺҶеҪ“еүҚйҳҹеҲ—дёӯжүҖжңүиҠӮзӮ№
        for (int i = 0; i < size; i++) {
            String curt = queue.poll();
            // и®°еҪ•еҪ“еүҚиҠӮзӮ№еҲ°startзҡ„жңҖзҹӯи·қзҰ»
            map.get(distance).add(curt);
            // еҰӮжһңйҒҮеҲ°зӣ®ж ҮиҠӮзӮ№пјҢйҖҖеҮәbfs
            if (curt.equals(endWord))
                return distance;
            // еҜ»жүҫиҜҘиҠӮзӮ№иғҪиө°еҲ°зҡ„жүҖжңүиҠӮзӮ№
            List<String> neighbors = expanding(curt.toCharArray());
            for (String neighbor : neighbors) {
                if (dict.contains(neighbor) && !visited.contains(neighbor)) {
                    // е°ҶжңӘи®ҝй—®зҡ„жңүж•ҲиҠӮзӮ№еҠ е…ҘйҳҹеҲ—
                    queue.offer(neighbor);
                    visited.add(neighbor);
                }
            }
        }
    }
    return 0;
}

private void dfs(String curt, int distance, Map<Integer, Set<String>> map, List<String> path, List<List<String>> ans) {
    if (distance == 0) {
        // еҪ“distanceдёә0ж—¶пјҢиҜҙжҳҺеҲ°иҫҫstart
        // е°ҶеҪ“еүҚжүҫеҲ°зҡ„и·Ҝеҫ„еӯҳе…Ҙans
        ans.add(new ArrayList<>(path));
        return;
    }
    // иҺ·еҸ–жүҖжңүи·қзҰ»startдёәеҪ“еүҚdistanceзҡ„иҠӮзӮ№
    Set<String> set = map.get(distance);
    // иҺ·еҫ—еҪ“еүҚиҠӮзӮ№иғҪиө°еҲ°зҡ„жүҖжңүиҠӮзӮ№
    List<String> neighbors = expanding(curt.toCharArray());
    for (String neighbor : neighbors) {
        // д»ҺиҝҷдәӣиҠӮзӮ№дёӯпјҢжүҫеҲ°еҗ‘startи·қзҰ»жӣҙиҝ‘дёҖжӯҘзҡ„иҠӮзӮ№
        if (set.contains(neighbor)) {
            // е°ҶиҜҘиҠӮзӮ№ж·»еҠ иҝӣpath
            path.add(neighbor);
            // йҖ’еҪ’жү§иЎҢdfs
            dfs(neighbor, distance - 1, map, path, ans);
            // еӣһжәҜзҠ¶жҖҒпјҢз§»йҷӨиҠӮзӮ№
            path.remove(path.size() - 1);
        }
    }
}

// дёҺжЎҲдҫӢ3зҡ„expandingеҮҪж•°зӣёеҗҢ
private List<String> expanding(char[] word) {
    List<String> neighbors = new ArrayList<>();
    // еңЁеҚ•иҜҚзҡ„жҜҸдёҖдёӘдҪҚзҪ®
    for (int i = 0; i < word.length; i++) {
        char tmp = word[i];
        // з”Ёa-zеҫӘзҺҜжӣҝжҚўиҜҘдҪҚзҪ®зҡ„еӯ—з¬Ұ
        for (char j = 'a'; j <= 'z'; j++) {
            word[i] = j;
            neighbors.add(new String(word));
        }
        word[i] = tmp;
    }
    return neighbors;
}
